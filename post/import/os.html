      Βιβλιοθήκη OS

      Το os στην Python παρέχει πολλές λειτουργίες για τον έλεγχο και τη διαχείριση του λειτουργικού συστήματος.

      Εισαγωγή Βιβλιοθήκης

      import os

      Εμφάνιση τρέχων κατάλογος

      import os
directory = os.getcwd()
print("Ο τρέχων κατάλογος είναι:", directory)

      Λίστα αρχείων και φακέλων σε έναν κατάλογο

      import os
dir_path = "./"
# Λίστα αρχείων και φακέλων στον καθορισμένο κατάλογο
files = os.listdir(dir_path)
for item in files:
    print(item)

      Δημιουργία φακέλου

      import os
dir_name = "name_dir"
# Δημιουργία του φακέλου στον τρέχοντα κατάλογο
os.mkdir(dir_name)

      Διαγραφή φακέλου (με περιεχόμενο)

      import os
dir_name = "name_dir"
# Διαγραφή του φακέλου και του περιεχομένου του
os.rmdir(dir_name)

      Ελέγχος ύπαρξης ενός αρχείου ή φακέλου

      import os
path_to_check = "./name_dir"
if os.path.exists(path_to_check):
    print(f"{path_to_check} υπάρχει.")
else:
    print(f"{path_to_check} δεν υπάρχει.")

      Αλλαγή του ονόματος ενός αρχείου ή φακέλου

      import os
old_name = "old_name.txt"
new_name = "new_name.txt"
os.rename(old_name, new_name)

      Εκτέλεση ενός εξωτερικού προγράμματος ή command

      import os
command_to_run = "dir"  # Παράδειγμα ενός Windows command
result = os.system(command_to_run)
# Ελέγξτε το αποτέλεσμα της εκτέλεσης
if result == 0:
    print(f"Η εντολή {command_to_run} εκτελέστηκε με επιτυχία.")
else:
    print(f"Η εντολή {command_to_run} απέτυχε με κωδικό εξόδου {result}.")

      Εύρεση του path ενός αρχείου ή Φακέλου

      import os
name = "name_dir"
path = os.path.abspath(name)
print(f"Το path του {name} είναι: {path}")

      Διεργασίες

      Windows

      import os
# Εκτελούμε την εντολή "tasklist" και αποθηκεύουμε τα αποτελέσματα σε ένα string
process_list = os.popen("tasklist").read()
# Εκτυπώνουμε τη λίστα των διεργασιών
print(process_list)
